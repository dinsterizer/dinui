## Usage

```tsx
import { Popover, PopoverContent, PopoverTrigger } from '@dinui/react/popover'
```

```tsx
<Popover>
  <PopoverTrigger>Open</PopoverTrigger>
  <PopoverContent>Place content for the popover here.</PopoverContent>
</Popover>
```
