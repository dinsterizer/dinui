## Installation

import InstallationGuide from '@web/components/installation-guide.astro'

<InstallationGuide path="ui/sonner">
<li>
  <p>Add the Toaster component to your root layout</p>

```tsx
import { Toaster } from '@dinui/react/sonner'

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <head />
      <body>
        <main>{children}</main>
        <Toaster />
      </body>
    </html>
  )
}
```

</li>
</InstallationGuide>

## Usage

```tsx
import { toast } from '@dinui/react/sonner'
```

```tsx
toast('Event has been created.')
```
