## Installation

<InstallationGuide path="ui/toast">
<li>
  <p>Add the Toaster component to your root layout</p>

```tsx
import { Toaster } from '@dinui/react/toast'

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <head />
      <body>
        <main>{children}</main>
        <Toaster />
      </body>
    </html>
  )
}
```
